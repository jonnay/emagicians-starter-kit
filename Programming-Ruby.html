3.4 Mode setup

Thanks to the emacs starter kit classic for the list of ode files list.

https://github.com/technomancy/emacs-starter-kit/blob/v2/modules/starter-kit-ruby.el 

(use-package ruby-mode
  :interpreter "ruby"
  :mode 
  (("\\.rake$" . ruby-mode)
   ("\\.task$" . ruby-mode)
   ("\\.thor$" . ruby-mode)
   ("\\.gemspec$" . ruby-mode)
   ("\\.rb$" . ruby-mode)
   ("\\.ru$" . ruby-mode)
   ("Rakefile$" . ruby-mode)
   ("Thorfile$" . ruby-mode)
   ("Gemfile$" . ruby-mode)
   ("Capfile$" . ruby-mode)
   ("Vagrantfile$" . ruby-mode))
  :init
  (use-package ruby-tools)
  (emagician/minor-in-major-mode ruby-tools-mode ruby-mode-hook)
  :config
  (ruby-block-mode t))
